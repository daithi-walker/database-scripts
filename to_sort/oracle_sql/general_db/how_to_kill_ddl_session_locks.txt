SELECT   'ALTER SYSTEM KILL SESSION '''||a.sid||','||c.serial#||''' IMMEDIATE;' sql
,        a.sid
,        c.serial#
,        a.job
,        a.failures
,        to_char(a.this_date, 'mm/dd/yyyy hh:mi pm') startdatetime
,        b.what
FROM     dba_jobs_running a
,        dba_jobs b
,        v$session c
WHERE    1=1
AND      a.job = b.job
AND      a.sid = c.sid
ORDER BY a.this_date;
